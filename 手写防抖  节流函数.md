# 手写防抖  节流函数

## 防抖debounce 函数   -  不会没按一次就执行一次(延迟)，连续点击，以最后一次点击时的时间为准，间隔一会执行

### 防抖引用场景：1.输入框中频繁的输入内容，搜索或者提交信息。2.频繁的点击按钮，触发某个事件 3.监听浏览器滚动事件，完成某些特定操作  4.用户缩放浏览器的resize事件

生活中例子：1分钟之内没有同学继续提问，就会下课

## 手写防抖

~~~js
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
</head>
<body>
    <input class="search" type="text">
    <script>
        var search = document.querySelector(".search");
        var conunter = 1;
        function debounce(fn,num=1000){
            // 定义一个定时器，保存上一次的定时器
            let time = null
            return function(){
                if(time){
                    // 取消上一次的定时器
                    clearTimeout(time)
                }
                // 延迟执行
                time = setTimeout(()=>{
                    // 外部传入的真正要执行的函数
                    fn()
                    conunter++
                },num)

                 }
                        
                }
            
        a = debounce(function(){
            console.log("发送"+conunter+ "网络请求")
        })
        search.oninput = a
    </script>
</body>
</html>
~~~

## 立即执行(连续输入第一次都会立即执行)

~~~js
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
</head>
<body>
    <input class="search" type="text">
    <script>
        var search = document.querySelector(".search");
        var conunter = 1;
        // immditate：true  表示每次连续输入第一个字母都会触发一次
        function debounce(fn,num=1000, immediate = false){
            let time = null
            // 控制连续第一次输入都执行便令
            let isInvode = false
            return function(...args){
                if(time){
                    clearTimeout(time)
                }
                // 判断是否需要立即执行
                if(immediate && !isInvode){
                    // 立即执行 箭头函数this指向上一级debounce ,
                    fn.apply(this, args)
                    conunter++
                    isInvode = true
                }
                else{
                    time = setTimeout(()=>{
                    fn.apply(this,args)
                    conunter++
                    isInvode = false
                },num)
                }
                

                 }
                        
                }
            
        a = debounce(function(){
            console.log("发送"+conunter+ "网络请求")
        },2000, true)
        search.oninput = a
    </script>
</body>
</html>


~~~









# 节流 

### 节流  -   发射子弹，不管你按的多快，子弹还是间隔一样的时间出来

### 节流引用场景：1.监听页面的滚动事件  2.鼠标移动事件  3.用户频繁点击按钮操作   游戏中的一些设计

生活中例子：一分钟之内，不管有多少同学来问问题，我只回答一个问题





## 手动实现节流

~~~js
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
</head>
<body>
    <input class="search" type="text">
    <script>
        var search = document.querySelector(".search");
        var conunter = 1;
        let lastTime = 0
        function throttle(fn, interval){
            const _throttle = function(){
                const nowTime = new Date().getTime()
                const remainTime = interval - (nowTime - lastTime)
                if(remainTime <= 0){
                    fn()
                    conunter++
                    lastTime = nowTime
                }

                
            }
            return _throttle
        }
            
        a = throttle(function(){
            console.log("发送"+conunter+ "网络请求")
        },2000)
        search.oninput = a
    </script>
</body>
</html>

~~~



## 不让第一次输入触发事件

~~~js
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
</head>
<body>
    <input class="search" type="text">
    <script>
        var search = document.querySelector(".search");
        var conunter = 1;
        function throttle(fn, interval, options = {leading:true, trailing: false}){
            let lastTime = 0
            const {leading, trailing} = options
            const _throttle = function(){
                const nowTime = new Date().getTime()

                // 第一次触发的时候不执行
                if(!lastTime && !leading){
                    lastTime = nowTime
                }
                const remainTime = interval - (nowTime - lastTime)
                if(remainTime <= 0){
                    fn()
                    conunter++
                    lastTime = nowTime
                }

                
            }
            return _throttle
        }
            
        a = throttle(function(){
            console.log("发送"+conunter+ "网络请求")
        },2000,{leading:false})
        search.oninput = a
    </script>
</body>
</html>
~~~
